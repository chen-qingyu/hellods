184void MatrixGraph_BFS(MatrixGraph* self, MatrixGraphVertex startV, void (*p_visit)(MatrixGraphVertex V))
185{

209bool MatrixGraph_Dijkstra(const MatrixGraph* self, MatrixGraphEdge dist[], MatrixGraphVertex path[], MatrixGraphVertex startV)
210{
211 clean_visited_flag();
212
213 // init array state
214 for (MatrixGraphVertex V1 = 0; V1 < self->vertex_num; V1++)
215 {
216 dist[V1] = self->matrix[startV][V1];
217 path[V1] = dist[V1] < NO_PATH ? startV : NOT_FOUND;
218 }
219
220 dist[startV] = 0;
221 visited[startV] = true;
222
223 while (1)
224 {
225 MatrixGraphVertex V1 = find_min_dist(self, dist);
226 if (V1 == NOT_FOUND)
227 {
228 break;
229 }
230 visited[V1] = true;
231 for (MatrixGraphVertex V2 = 0; V2 < self->vertex_num; V2++)
232 {
233 if (visited[V2] == false && self->matrix[V1][V2] < NO_PATH)
234 {
235 if (self->matrix[V1][V2] < 0)
236 {
237 return false;
238 }
239 if (dist[V1] + self->matrix[V1][V2] < dist[V2])
240 {
241 dist[V2] = dist[V1] + self->matrix[V1][V2];
242 path[V2] = V1;
243 }
244 }
245 }
246 }
247
248 return true;
249}

251bool MatrixGraph_Floyd(const MatrixGraph* self, MatrixGraphEdge dist[][VERTEX_NUMBER], MatrixGraphVertex path[][VERTEX_NUMBER])
252{
253 for (MatrixGraphVertex i = 0; i < self->vertex_num; i++)
254 {
255 for (MatrixGraphVertex j = 0; j < self->vertex_num; j++)
256 {
257 dist[i][j] = self->matrix[i][j];
258 path[i][j] = NOT_FOUND;

261
262 for (MatrixGraphVertex k = 0; k < self->vertex_num; k++)
263 {
264 for (MatrixGraphVertex i = 0; i < self->vertex_num; i++)
265 {
266 for (MatrixGraphVertex j = 0; j < self->vertex_num; j++)
267 {
268 if (dist[i][k] + dist[k][j] < dist[i][j])

270 dist[i][j] = dist[i][k] + dist[k][j];
271 if (i == j && dist[i][j] < 0)

273 return false;

275 path[i][j] = k;

278 }
